add_namespace = election

#Actual Election
country_event = {
	id = election.1
	title = election.1.t
	desc = election.1.d
	picture = GFX_report_event_election_vote
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { original_tag = WST }
	}
	
	option = {	#The Ruling Party
		name = election.1.o1
		
		trigger = {
			OR = {
				AND = { has_government = democratic democratic > 0.5 }
				AND = { has_government = communism communism > 0.5 }
				AND = { has_government = fascism fascism > 0.5 }
				AND = { has_government = neutrality neutrality > 0.5 }
				AND = {
					democratic < 0.5
					communism < 0.5
					fascism < 0.5
					neutrality < 0.5
				}
			}
		}
		
		ai_chance = {
			factor = 1
		}
		
		add_political_power = 20
	}
	
	option = {	#Democrats
		name = election.1.o2
		ai_chance = {
			factor = 1
		}
		trigger = {
			NOT = { has_government = democratic }
			democratic > 0.5
		}
		
		set_politics = {
			ruling_party = democratic
			elections_allowed = yes
		}
	}
	
	option = {	#Anarchists
		name = election.1.o3
		trigger = {
			NOT = { has_government = communism }
			communism > 0.5
		}
		ai_chance = {
			factor = 1
		}
		
		set_politics = {
			ruling_party = communism
			elections_allowed = no
		}
	}
	
	option = {	#Ultranationalists
		name = election.1.o4
		ai_chance = {
			factor = 1
		}
		trigger = {
			NOT = { has_government = fascism }
			fascism > 0.5
		}
		
		set_politics = {
			ruling_party = fascism
			elections_allowed = no
		}
	}
	
	option = {	#Autocrats
		name = election.1.o5
		trigger = {
			NOT = { has_government = neutrality }
			neutrality > 0.5
		}
		ai_chance = {
			factor = 1
		}
		
		set_politics = {
			ruling_party = neutrality
			elections_allowed = no
		}
	}
}
